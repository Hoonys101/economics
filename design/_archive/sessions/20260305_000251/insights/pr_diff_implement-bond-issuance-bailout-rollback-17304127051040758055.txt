+
+    request = AssetBuyoutRequestDTO(seller_id=1, inventory={"A": 10}, market_prices={"A": 100})
+
+    result = AssetBuyoutResultDTO(success=True, total_paid_pennies=1000, items_acquired={"A": 10}, buyer_id=2, transaction_id="test_bailout")
+
